USE [Pizza DB]

Select * From [pizza_sales excel]

SELECT SUM(total_price) AS Total_Revenue FROM [pizza_sales excel]	

SELECT (SUM(total_price) / COUNT(DISTINCT order_id)) AS Avg_order_Value FROM [pizza_sales excel]

-- Daily Trend for Total Orders

SELECT DATENAME(DW, order_date) AS order_day, COUNT(DISTINCT order_id) AS total_orders 
FROM [pizza_sales excel]
GROUP BY DATENAME(DW, order_date)

-- Hourly Trend for Orders
SELECT DATEPART(HOUR, order_time) as order_hours, COUNT(DISTINCT order_id) as total_orders
from [pizza_sales excel]
group by DATEPART(HOUR, order_time)
order by DATEPART(HOUR, order_time)
